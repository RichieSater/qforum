                log.info('CAPITULATION')  
                action = -1  
                hold = 1  
        if mode == prev_mode == 2:  
            if ma2 < ma90:  
                # hold = 0  
                if holding_currency:  
                    log.info('DESPAIR')  
                    action = 1  
                    hold = 1  
            if ((ma2 > float(1.1) * ma90) and  
                    (ma2 > ma60)):  
                # hold = 0  
                if holding_assets:  
                    log.info('FINAL RUN OF THE BULLS - B')  
                    action = -1  
                    hold = 2

        # MODE 3 - RED DRAGON  
        if ((mode == 3) and (prev_mode != 3)):  
            hold = 0  
            if holding_assets:  
                log.info('RED DRAGON')  
                action = -1  
                hold = 5  
        if mode == prev_mode == 3:  
            signal_3 = mode_3(context)  
            if signal_3 == 1:  
                hold = 5  
                if holding_currency:  
                    log.info('PURGATORY')  
                    action = 1  
                    hold = 5  
            elif signal_3 == -1:  
                hold = 0  
                if holding_assets:  
                    log.info('DEEPER')  
                    action = -1  
                    hold = 1

        # MODE 4 - CAT BOUNCE  
        if ((mode == 4) and (prev_mode != 4)):  
            hold = 0  
            if holding_currency:  
                log.info('HONEY')  
                action = 1  
                hold = 2  
        if mode == prev_mode == 4:  
            if ((ma2 < float(1.05) * ma30) and  
                (ma2 > ma60) and  
                    (ma90 > float(1.003) * ma90i)):  
                hold = 1  
                if holding_assets:  
                    log.info('KARMA')  
                    action = -1  
                    hold = 1  
            if ((ma2 < ma90) and  
                (ma90 < ma60) and  
                (ma90 < ma30) and  
                    (ma90 > ma90i)):  
                hold = 0  
                if holding_currency:  
                    log.info('GOLD')  
                    action = 1  
                    hold = 1

    context.action = action  
    context.prev_mode = context.mode

def trade(context):  
    if abs(context.action) ==1:  
        log.info(context.mode)    

    if context.action == 1:  
        if context.holding_currency:  
            order_percent(SECURITY, 1)  
    if context.action == -1:  
        if context.holding_assets:  
            order_percent(SECURITY, -1)

    indicators(context)  
    chart(context)  
    think(context)  
    trade(context)
